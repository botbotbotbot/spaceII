drawSymbol(EQdata)
	connectorRadians = experimentally determined, the amount of radians reserved for connecting 
					   the beginning of the symbol to the end
	center = (width/2,height/2)
	setFillColor(EQdata)
	n = length(EQdata)/(2*pi-connectorRadians)
	for i in range(n):
		ray(0,EQdata[i],EQdata[i+1],i,i+1)
	ray(center,EQdata[i+1],EQdata[0],2*pi-connectorRadians,0) // 0 = 2*pi
